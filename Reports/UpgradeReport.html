Voor avondbezorging is vertreklocatie:cluster 1:1, voor daglevering kan dit anders zijn
*/

	[VoertuigProfiel] INT NOT NULL, /* Welk voertuig profiel in OSRM gebruikt dient te worden */

	[BezorgenTot] VARCHAR(6),
	[BeschikbaarheidZendingen] VARCHAR(6),
	[VertrekRitten] INT, 
	[UitersteRitduur] INT NULL, /* Maximale ritduur */
	[VertrekTijdstip] VARCHAR(6) NULL, /* Tijdstip vanaf hoe laat kan worden vertrokken vanaf de vertreklocatie */
	[AantalVoertuigen] INT NULL, /* Maximaal aantal voertuigen dat ingezet mag worden in de planning (voor avondritten op bijvoorbeeld 200 zetten) */
	[CapaciteitVoertuigenVolume] INT NULL, /* Dit is de totale capaciteit van het voertuig (hoeveel hij mee kan nemen). De eenheid die hier wordt gebruikt, moet overeenkomen met de eenheid dit per order wordt opgegeven. Dus voor avond in m^3. Effectief laadvermogen van een NLI geschikte bus is ongeveer 6,5 m^3 */
	[CapaciteitVoertuigenGewicht] INT NULL, /* Dit is de totale capaciteit van het voertuig (hoeveel hij mee kan nemen). De eenheid die hier wordt gebruikt, moet overeenkomen met de eenheid dit per order wordt opgegeven. Dus voor avond in kg. Effectief laadvermogen van een NLI geschikte bus is ongeveer 1000 kg */
	[RijtijdFactor] FLOAT NULL, /* Indien nodig kan de rijtijd van alle afstanden die gebruikt worden in de planning met deze factor opgehoogd worden. */
	[Plaatjes] INT NULL, /* Boolean om aan te geven of je wel of geen plaatjes wilt laten maken */
	[VoorbereidingstijdMinuten] INT NULL, /* De tijd die de chauffeur bezig is met laden etc.. Voordat hij kan vertrekken (bij avond 15 minuten) */
	[DebriefTijdMinuten] INT NULL, /* De tijd die de chauffeur bezig is bij de debrief (bij avond 15 minuten) */

	[DistributionChannelId] INT,
	/*
	Vertrek ritten ("zo vroeg mogelijk" of "zoveel mogelijk tegelijk" )
	1 = zo vroeg mogelijk
	2 = zoveel mogelijk tegelijk
	3 = zo laat mogelijk
	*/
	[Actief] BIT NOT NULL,
	FOREIGN KEY ([DistributionChannelId]) REFERENCES [Configuratie].[DistributionChannels],
	FOREIGN KEY ([PlanningConfiguratieId]) REFERENCES [Configuratie].[PlanningConfiguraties],
